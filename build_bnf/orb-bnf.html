CompilationUnit ::= PackageDecl ( UsingDecl )* ( ConstDecl )* ( PackagePropertyDecl )* ( PackageVariableDecl )* ( PackageFunctionDecl )*
PackageDecl ::= "package" ComplexName ";"
UsingDecl ::= "using" ComplexName ( "." "*" )? ( "alias" Name )? ";"
ConstDecl ::= Modifiers "const" Name "=" Exp ";"
PackagePropertyDecl ::= "property" ( "readonly" | "readwrite" | "writeonly" )? Name "=" Exp ";"
PackageVariableDecl ::= "local" Name ( "=" Exp )? ";"
PackageFunctionDecl ::= Modifiers "function" Name "(" ( FormalParams )? ")" ( "return" "(" ResultParams ")" )? "is" ScopeBlock "end"
FormalParams ::= FormalParam ( "," FormalParam )*
FormalParam ::= ( Name ( "=" Exp )? | "..." Name )
ResultParams ::= ResultParam ( "," ResultParam )*
ResultParam ::= ( Name | "..." Name )
Name ::= <IDENTIFIER>
ComplexName ::= Name ( "." Name )*
ComplexNameList ::= ComplexName ( "," ComplexName )*
Modifiers ::= ( Modifier )+
Modifier ::= ( "public" | "local" )
Literal ::= <INTEGER_LITERAL>
| <FLOATING_POINT_LITERAL>
| <CHARACTER_LITERAL>
| <STRING_LITERAL>
| BooleanLiteral
| NullLiteral
BooleanLiteral ::= "true"
| "false"
NullLiteral ::= "nil"
ScopeBlock ::= ( LocalVariableDecl )* ( Stmt )* ( EndStmt )?
LocalVariableDecl ::= "local" VariableDecl ( "," VariableDecl )* ";"
VariableDecl ::= Name ( "=" Exp )?
Stmt ::= ( FunctionCallStat | AssignmentStat | IfStat | ForStat | WhileStat | CaseStat )
FunctionCallStat ::= PrimaryExpression Arguments ";"
AssignmentStat ::= PrimaryExpression AssignmentOperator Exp ";"
IfStat ::= "if" Exp "then" ScopeBlock ( "elif" Exp "then" ScopeBlock )* ( "else" ScopeBlock )? "end"
ForStat ::= "for" Name "=" Exp "," Exp ( "," Exp )? "alias" Name "do" ScopeBlock "end"
WhileStat ::= "while" Exp "alias" Name "do" ScopeBlock "end"
CaseStat ::= "case" PrimaryExpression ( "when" LiteralList "then" ScopeBlock )* ( "default" ScopeBlock )? "end"
LiteralList ::= Literal ( "," Literal )*
EndStmt ::= ( ReturnStat | ContinueStat | BreakStat )
ReturnStat ::= "return" ";"
ContinueStat ::= "continue" ( Name )? ";"
BreakStat ::= "break" ( Name )? ";"
PrimaryExpression ::= PrimaryPrefix ( PrimarySuffix )*
PrimaryPrefix ::= ( Literal | "this" | Name )
PrimarySuffix ::= ( "[" Exp ( ":" Exp )? ( "," Exp ( ":" Exp )? )? "]" | "." Name | Arguments )
Arguments ::= "(" ( ExpList )? ")"
ExpList ::= Exp ( "," Exp )*
Exp ::= PrimaryExpression ( ( "++" | "--" ) )?
| ConditionalExpression
| ArrayDecl
| DictDecl
ArrayDecl ::= "[" ( ExpList )? "]"
DictDecl ::= "{" ( PairDecl ( "," PairDecl )* )? "}"
PairDecl ::= Exp ":" Exp
AssignmentOperator ::= "="
| "*="
| "/="
| "%="
| "+="
| "-="
| "&="
| "^="
| "|="
ConditionalExpression ::= ConditionalOrExpression ( "?" Exp ":" ConditionalExpression )?
ConditionalOrExpression ::= ConditionalAndExpression ( ( "||" | "or" ) ConditionalAndExpression )*
ConditionalAndExpression ::= InclusiveOrExpression ( ( "&&" | "and" ) InclusiveOrExpression )*
InclusiveOrExpression ::= ExclusiveOrExpression ( "|" ExclusiveOrExpression )*
ExclusiveOrExpression ::= AndExpression ( "^" AndExpression )*
AndExpression ::= EqualityExpression ( "&" EqualityExpression )*
EqualityExpression ::= InstanceOfExpression ( ( "==" | "~=" ) InstanceOfExpression )*
InstanceOfExpression ::= RelationalExpression ( "typeof" ComplexName )?
RelationalExpression ::= ShiftExpression ( ( "<" | ">" | "<=" | ">=" ) ShiftExpression )*
ShiftExpression ::= AdditiveExpression ( ( "<<" | ">>" | ">>>" ) AdditiveExpression )*
AdditiveExpression ::= MultiplicativeExpression ( ( "+" | "-" ) MultiplicativeExpression )*
MultiplicativeExpression ::= UnaryExpression ( ( "*" | "/" | "%" ) UnaryExpression )*
UnaryExpression ::= ( "+" | "-" ) PrimaryExpression
| UnaryExpressionNotPlusMinus
UnaryExpressionNotPlusMinus ::= ( "~" | "!" | "#" ) PrimaryExpression
